Updating the Elements of a List

my_list = ['apple', 'banana', 'cherry']

print(f"Original list: {my_list}")
 
# Updating a single element

# Updating a slice of the list

Membership in Lists

fruits = ['apple', 'banana', 'cherry']
 
# Check if 'apple' is in the list

is_apple_present = 'apple' in fruits

print(f"Is 'apple' in the list? {is_apple_present}")
 
# Check if 'grape' is not in the list

is_grape_absent = 'grape' not in fruits

print(f"Is 'grape' not in the list? {is_grape_absent}")
 
Aliasing and Cloning Lists

list_a = [1, 2, 3]
 
# Aliasing (both variables point to the same list object)

list_b = list_a

list_b[0] = 99

print(f"Original list_a after aliasing: {list_a}")

print(f"Alias list_b: {list_b}")
 
# Cloning (creating a new list object with the same elements)

list_c = list_a[:]  # or list_c = list_a.copy()

list_c[0] = 100

print(f"\nOriginal list_a after cloning: {list_a}")

print(f"Cloned list_c: {list_c}")
 
Methods to Process Lists

my_list = [10, 20, 30]
 
# append() - adds an element to the end

my_list.append(40)

print(f"After append(40): {my_list}")
 
# insert() - adds an element at a specific index

my_list.insert(1, 15)

print(f"After insert(1, 15): {my_list}")
 
# pop() - removes and returns an element at a specific index

popped_element = my_list.pop(2)

print(f"Popped element at index 2: {popped_element}, List is now: {my_list}")
 
# remove() - removes the first occurrence of a value

my_list.remove(40)

print(f"After remove(40): {my_list}")
 
# reverse() - reverses the list in-place

my_list.reverse()

print(f"After reverse(): {my_list}")

Sorting the List Elements

numbers = [5, 2, 8, 1, 9]
 
# sort() - sorts the list in-place

numbers.sort()

print(f"Sorted list using sort(): {numbers}")
 
# sorted() - returns a new sorted list

original_list = [5, 2, 8, 1, 9]

new_sorted_list = sorted(original_list)

print(f"Original list: {original_list}")

print(f"New sorted list using sorted(): {new_sorted_list}")

Exercise : Finding Common Elements in Two Lists

list1 = [1, 2, 3, 4]

# Using sets for efficiency

set1 = set(list1)

common_elements_set = list(set1.intersection(set2))
