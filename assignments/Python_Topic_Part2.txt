# Length of a String
# The `len()` function returns the number of characters in a string.
s = "Python"
print(len(s)) # Output: 6
 
# ---
 
# Indexing in Strings
# Strings are sequences, and you can access individual characters using their position (index).
# Indexing starts from 0 for the first character. Negative indices count from the end, starting with -1.
s = "Python"
print(s[0])   # Output: P (the first character)
print(s[-1])  # Output: n (the last character)
 
# ---
 
# Slicing the Strings
# Slicing extracts a part of a string. The syntax is `[start:end:step]`.
# `start` is the starting index (inclusive), `end` is the ending index (exclusive), and `step` is the increment.
s = "Python"
print(s[2:5])   # Output: tho (characters from index 2 up to, but not including, 5)
print(s[:3])    # Output: Pyt (from the beginning up to index 3)
print(s[3:])    # Output: hon (from index 3 to the end)
print(s[::2])   # Output: Pto (every second character)
print(s[::-1])  # Output: nohtyP (reverses the string)
print(s[1:] 	# Output: Python (Prints entire string)
